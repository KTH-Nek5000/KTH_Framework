728  subroutine elasolv (nel)
729 C
730 C Elastostatic solver for mesh deformation
731 C
732  include 'SIZE'

735  include 'MVGEOM'

737  include 'TSTEP'

799 
800  ifldt = ifield
801  ifield=1
802  if (ifheat) ifield=2
803  call dsavg(wx)
804  call dsavg(wy)
805  call dsavg(wz)
806  ifield = ifldt
807 
808 c if (istep.gt.1) then
809 c ifldx = ifield
810 c ifield = 1

813 c return
814 c endif
815
