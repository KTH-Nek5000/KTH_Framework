857  subroutine calc (diag,upper,d,e,n,dmax,dmin)
858 c
859  dimension diag(n),upper(n)
860  dimension d(n),e(n)
861 c
862  call copy (d,diag ,n)
863  call copy (e,upper,n)
864 c
865  do 15 l=1,n
866  iter = 0
867 c
868  1 do 12 m=l,n-1
869  dd = abs( d(m) ) + abs( d(m+1) )
870  if ( abs(e(m)) + dd .eq. dd ) goto 2
871  12 continue
872 c
873  m = n
874  2 if ( m .ne. l ) then
875 c
876  if ( iter .eq. 30 ) then
877  write (6,*) 'too many iterations'
878  return
879  endif
880 c
881  iter = iter + 1
882  g = ( d(l+1) - d(l) ) / ( 2.0 * e(l) )
883  r = sqrt( g**2 + 1.0 )
884 c
885 c sign is defined as a(2) * abs( a(1) )
886 c
887  g = d(m) - d(l) + e(l)/(g+sign(r,g))
888  s = 1.0
889  c = 1.0
890  p = 0.0
891 c
892  do 14 i = m-1,l,-1
893  f = s * e(i)
894  b = c * e(i)
895  if ( abs(f) .ge. abs(g) ) then
896  c = g/f
897  r = sqrt( c**2 + 1.0 )
898  e(i+1) = f*r
899  s = 1.0/r
900  c = c*s
901  else
902  s = f/g
903  r = sqrt( s**2 + 1.0 )
904  e(i+1) = g*r
905  c = 1.0 / r
906  s = s * c
907  endif
908 c
909  g = d(i+1) - p
910  r = ( d(i) - g ) * s + 2.0 * c * b
911  p = s * r
912  d(i+1) = g + p
913  g = c*r - b
914  14 continue
915 c
916  d(l) = d(l) - p
917  e(l) = g
918  e(m) = 0.0
919  goto 1
920 c
921  endif
922 c
923  15 continue
924 c
925  dmax = 0.0
926  dmin = d(1)
927 c
928  do 40 i = 1 , n
929  dmax = abs( max( d(i) , dmax ) )
930  dmin = abs( min( d(i) , dmin ) )
931  40 continue
932 c
933  return
