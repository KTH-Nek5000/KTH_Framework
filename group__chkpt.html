Detailed Description

Checkpointing routines for toolbox.

This is the main interface to I/O checkpointing.

Global interface dependency:

  3. chkpts_init; provided by submodule
  4. chkpts_read; provided by submodule
  5. chkpts_write; provided by submodule
Module interface usage:
!======================================================================
subroutine userchk
include 'TSTEP'
if (istep.eq.0) then
! start framework
call frame_start
endif
! monitor simulation
call frame_monitor
! save/load files for full-restart
call chkpt_main
! finalise framework
if (istep.eq.nsteps.or.lastep.eq.1) then
call frame_end
endif
return
end
!======================================================================
subroutine frame_usr_register
implicit none
include 'SIZE'
include 'FRAMELP'
!-----------------------------------------------------------------------
! register modules
call chkpt_register
return
end subroutine
!======================================================================
subroutine frame_usr_init
implicit none
include 'SIZE'
include 'FRAMELP'
!-----------------------------------------------------------------------
! initialise modules
call chkpt_init
return
end subroutine
!======================================================================
subroutine frame_usr_end
implicit none
include 'SIZE'
include 'FRAMELP'
!-----------------------------------------------------------------------
return
end subroutine
